  31.         for ( i = 0; i < Zips.Length; i++ )
  32.         {
  33.                 if (Zips[i].ZipName ~= ZipName)
  34.                 {
  35.                         ZipFound = True;

  37.                 }
  38.         }
  39.  
  40.         if (ZipFound) Zips[i] = Zip;
  41.         else Zips[Zips.Length] = Zip;
  42.  
  43.         SaveConfig();
  44. }

  61.                 if (Zips[i].ZipName ~= ZipName)
